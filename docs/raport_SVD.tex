\subsection{Descompunerea dupƒГ valori singulare}

\begin{definitie}
Fie $A \in \mathbb{R}^{m \times n}$. O descompunere dupƒГ valori singulare a matricei $A$ este
\[

\]

\[
  \sigma_1 \geq \sigma_2 \geq \cdots \geq \sigma_p \geq 0,\quad p=\min(m,n).
\]

\end{definitie}

\begin{teorema}[Existen»Ыa SVD]
Orice matrice realƒГ $A \in \mathbb{R}^{m \times n}$ admite o descompunere SVD.
\end{teorema}

\begin{proof}

\[
  A^T A = V \Lambda V^T,
\]
unde $\Lambda=\mathrm{diag}(\lambda_1,\ldots,\lambda_n)$ »Щi $\lambda_i \geq 0$. Definim
\[
  \sigma_i=\sqrt{\lambda_i}.
\]
Pentru $\sigma_i>0$, vectorii singulari st√Ґngi sunt
\[
  u_i=\frac{A v_i}{\sigma_i}.
\]
Ace»Щtia sunt ortonorma»Ыi deoarece
\[
  \langle u_i,u_j\rangle=
  \frac{v_i^T A^T A v_j}{\sigma_i\sigma_j}

  \frac{\lambda_j \langle v_i,v_j\rangle}{\sigma_i\sigma_j}

  \delta_{ij}.
\]
Vectorii lipsƒГ se completeazƒГ la o bazƒГ ortonormatƒГ, ob»Ыin√Ґndu-se matricea $U$. RezultƒГ factorizarea $A=U\Sigma V^T$.
\end{proof}

\subsection{Aproximarea de rang redus}

SVD-ul redus pƒГstreazƒГ doar primele $r$ valori singulare:
\[

\]

\begin{teorema}[Eckart--Young--Mirsky]

\[

\]

\subsection{Descompunerea Tucker »Щi HOSVD}

Descompunerea Tucker aproximeazƒГ tensorul $\mathcal{A}$ prin
\[
  \mathcal{A} \approx
  \mathcal{G}
  \times_1 U^{(1)}

\]
unde $\mathcal{G}$ este tensorul de bazƒГ, iar $U^{(n)}$ sunt matrici cu coloane ortonormate.

HOSVD calculeazƒГ aceste matrici aplic√Ґnd SVD pe fiecare matricizare:
\begin{algorithm}[H]
\caption{HOSVD}
\begin{algorithmic}[1]
\Require Tensor $\mathcal{A}$ »Щi ranguri $(R_1,R_2,R_3)$
\For{$n=1,2,3$}
  \State $A_{(n)} \leftarrow \mathrm{matricize}(\mathcal{A}, n)$
  \State $A_{(n)} \approx U_r^{(n)}\Sigma_r^{(n)}V_r^{(n)T}$ prin SVD redus
  \State Se pƒГstreazƒГ primele $R_n$ coloane din $U^{(n)}$
\EndFor
\State $\mathcal{G} \leftarrow \mathcal{A}\times_1 U^{(1)T}\times_2 U^{(2)T}\times_3 U^{(3)T}$
\Return $\mathcal{G}, U^{(1)}, U^{(2)}, U^{(3)}$
\end{algorithmic}
\end{algorithm}

Reconstruc»Ыia se face invers:
\[

  \mathcal{G}
  \times_1 U^{(1)}
  \times_2 U^{(2)}
  \times_3 U^{(3)}.
\]
